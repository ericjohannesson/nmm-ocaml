let auto_numbering_of_options (options : string list) : int -> int -> string = let rec aux (lst : string list) : int -> int -> string = match lst with |[] -> auto_numbering_default |"--numbering"::(s::_) -> ( try auto_numbering_of_string s with |Invalid_argument a -> let _ : unit = Debug_utils.print_warning ("WARNING: Invalid auto-numbering argument: \'" ^ a ^ "\'; using default") in auto_numbering_default ) |hd::tl -> aux tl in aux options
