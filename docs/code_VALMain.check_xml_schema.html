let check_xml_schema (path : string) : string = try let dtd:Dtd.dtd=Dtd.parse_file path in let _:Dtd.checked=Dtd.check dtd in String.concat " " [path;"is a well-defined xml-schema"] with Xml_light_errors.Dtd_check_error e -> raise (Error (String.concat " " [path;"->";"Xml_light_errors.Dtd_check_error:";Dtd.check_error e]))
