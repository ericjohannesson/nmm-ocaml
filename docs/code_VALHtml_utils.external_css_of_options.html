let external_css_of_options (options : string list) : string list = let rec aux (lst : string list) (acc : string list) = match lst with |[] -> acc |hd::tl -> match hd with |"--css" -> ( match tl with |uri::_ -> aux tl (uri::acc) |_ -> let _ : unit = Debug_utils.print_warning "WARNING: missing --css argument; using default" in aux tl acc ) |_ -> aux tl acc in aux options []
