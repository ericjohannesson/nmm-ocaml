let string_of_ts_ftn_ref (doc_settings : t_doc_settings) (ftn_table : t_ftn_table) (path : t_path) (ftn_ref : ts_ftn_ref) : string = match ftn_ref with |Cs_ftn_ref (id,i) -> let rec aux (table : t_ftn_table) : string = match table with |[] -> let _ : unit = Debug_utils.print_warning (String.concat "" [ "WARNING: id \'";string_of_tr_id id; "\' referenced in "; string_of_path doc_settings path; " does not exist or is out of scope."; ]) in "??" |(Ftn_entry_ref (table_ftn_ref, _, n, _))::tl -> if ftn_ref=table_ftn_ref then string_of_path doc_settings [FTN_NODE n] else aux tl |hd::tl -> aux tl in aux ftn_table
